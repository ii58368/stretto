drop database regi;
create database regi;
use regi;

create table person (id INT NOT NULL AUTO_INCREMENT,
                initials varchar(10) not null,
		id_instrument int NOT NULL,
		firstname varchar(30) NOT NULL,
		lastname varchar(30) NOT NULL,
		phone varchar(15) NOT NULL,
		email varchar(50) NOT NULL,
                status varchar(16) not null,
		table_ok char(1),
		comment varchar(300),
		primary key (id));
describe person;

create table shift (id_person int not null,
		id_project int not null,
                status int not null,
		comment varchar(300),
                primary key (id_person, id_project));
describe shift;

create table project (id INT NOT NULL AUTO_INCREMENT,
                name varchar(30) NOT NULL,
                semester varchar(1) NOT NULL,
                year int NOT NULL,
		id_def_responsible int not null,
		comment varchar(4096),
                primary key (id));
describe project;

create table resource (id_person int not null,
                id_plan int not null,
                status int NOT NULL,
                comment varchar(300),
                primary key (id_person, id_plan));
describe resource;

create table gresource (id_instrument int not null,
                id_plan int not null,
                status int NOT NULL,
                comment varchar(300),
                primary key (id_instrument, id_plan));
describe gresource;

create table plan (id INT NOT NULL AUTO_INCREMENT,
                date int not null,
		time varchar(16),
                tsort int not null,
		id_responsible int not null,
		id_location int not null,
		id_project int not null,
                comment varchar(300),
                primary key (id));
describe plan;

create table location (id int not null auto_increment,
                name varchar(64) not null,
                address varchar(256),
                url varchar(256),
                comment varchar(300),
                primary key (id));
describe location;

create table instruments (id int not null auto_increment,
                instrument varchar(30) NOT NULL,
                list_order INT NOT NULL,
                direction char(1) not null,
                comment varchar(300) NOT NULL,
                primary key (id));
describe instruments;
